import pandas as pd
import numpy as np
import lightgbm as lgb

# =============================
# 基本設定
# =============================
INITIAL_BANKROLL = 100000
KELLY_FRACTION = 0.05

LOW_ODDS = 20
HIGH_ODDS = 40
EDGE_THRESHOLD = 0.06
EXCLUDE_MONTHS = [6, 7, 8]

TEST_YEARS = [2024, 2025, 2026]

features = [
    "surface","distance","frame_no","weight_carried",
    "sex","age","field_size","past_race_count",
    "avg_finish_last5","days_since_last",
    "past_win_rate","is_first_race"
]

# =============================
# データ読み込み
# =============================
df = pd.read_pickle("training_ready.pkl")
df = df[df["odds"] > 0].copy()
df["race_date"] = pd.to_datetime(df["race_date"])
df["year"] = df["race_date"].dt.year
df["month"] = df["race_date"].dt.month

bankroll = INITIAL_BANKROLL
all_bets = []

# =============================
# 年次ウォークフォワード
# =============================
for test_year in TEST_YEARS:

    train_df = df[df["year"] < test_year].copy()
    test_df  = df[df["year"] == test_year].copy()

    if len(test_df) == 0:
        continue

    # -------------------------
    # group作成（レース単位）
    # -------------------------
    train_df = train_df.sort_values("race_id")
    test_df = test_df.sort_values("race_id")

    train_group = train_df.groupby("race_id").size().values
    test_group = test_df.groupby("race_id").size().values

    train_data = lgb.Dataset(
        train_df[features],
        label=train_df["is_win"],
        group=train_group
    )

    test_data = lgb.Dataset(
        test_df[features],
        label=test_df["is_win"],
        group=test_group,
        reference=train_data
    )

    params = {
        "objective": "lambdarank",
        "metric": "ndcg",
        "learning_rate": 0.03,
        "num_leaves": 63,
        "feature_fraction": 0.8,
        "bagging_fraction": 0.8,
        "bagging_freq": 1,
        "seed": 42,
        "verbose": -1
    }

    model = lgb.train(
        params,
        train_data,
        num_boost_round=2000,
        valid_sets=[test_data],
        callbacks=[lgb.early_stopping(200)]
    )

    # =============================
    # 予測スコア
    # =============================
    test_df["model_score"] = model.predict(test_df[features])

    # レース内softmaxで確率化
    def softmax(x):
        x = x - np.max(x)
        exp_x = np.exp(x)
        return exp_x / exp_x.sum()

    test_df["model_prob"] = test_df.groupby("race_id")["model_score"].transform(softmax)

    print(f"\n==== {test_year}年 確率合計確認 ====")
    print(test_df.groupby("race_id")["model_prob"].sum().describe())

    # =============================
    # 市場確率
    # =============================
    test_df["market_prob_raw"] = 1 / test_df["odds"]
    test_df["market_prob"] = test_df.groupby("race_id")["market_prob_raw"].transform(lambda x: x/x.sum())

    test_df["edge"] = test_df["model_prob"] - test_df["market_prob"]

    idx = test_df.groupby("race_id")["edge"].idxmax()
    bets = test_df.loc[idx].copy()

    bets = bets[
        (bets["odds"] > LOW_ODDS) &
        (bets["odds"] <= HIGH_ODDS) &
        (bets["edge"] > EDGE_THRESHOLD) &
        (~bets["month"].isin(EXCLUDE_MONTHS))
    ].copy()

    if len(bets) == 0:
        print(f"{test_year}年 → ベットなし")
        continue

    avg_pred = bets["model_prob"].mean()
    actual_hit = bets["is_win"].mean()

    print(f"\n==== {test_year}年 ベット統計 ====")
    print("ベット数:", len(bets))
    print("平均予測確率:", avg_pred)
    print("実際的中率:", actual_hit)
    print("差:", avg_pred - actual_hit)

    # 定額ROI
    bets["bet"] = 100
    bets["return"] = bets["bet"] * bets["odds"] * bets["is_win"]

    roi = bets["return"].sum() / bets["bet"].sum()
    print("定額ROI:", roi)

    # Kelly
    bets = bets.sort_values("race_date")

    for _, row in bets.iterrows():

        p = row["model_prob"]
        odds = row["odds"]
        b = odds - 1

        kelly = (p * odds - 1) / b
        kelly = max(kelly, 0)

        bet_fraction = kelly * KELLY_FRACTION
        bet_amount = bankroll * bet_fraction
        bet_amount = min(bet_amount, bankroll)

        if bet_amount <= 0:
            continue

        if row["is_win"] == 1:
            bankroll += bet_amount * (odds - 1)
        else:
            bankroll -= bet_amount

    all_bets.append(bets)

# =============================
# 全体結果
# =============================
if len(all_bets) > 0:

    all_bets = pd.concat(all_bets)

    print("\n========== 全体統計 ==========")
    print("総ベット数:", len(all_bets))
    print("全体的中率:", all_bets["is_win"].mean())
    print("平均オッズ:", all_bets["odds"].mean())
    print("平均edge:", all_bets["edge"].mean())

print("\n========== Kelly結果 ==========")
print("最終資金:", bankroll)
print("総増加率:", bankroll / INITIAL_BANKROLL)
